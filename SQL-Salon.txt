CREATE DATABASE Salon;

CREATE TABLE USUARIO(
	EMAIL VARCHAR(50) PRIMARY kEY,
	NOME VARCHAR(100) NOT NULL,
	SENHA VARCHAR(50) NOT NULL,
	TELEFONE VARCHAR(20),
	FOTOPERFIL TEXT,
	TIPO VARCHAR(1)
);
CREATE TABLE ATENDENTE(
	NOME VARCHAR(100) PRIMARY KEY,
	HORAINICIO VARCHAR(100) NOT NULL,
	HORAFIM VARCHAR(100) NOT NULL
);
CREATE TABLE SERVICO(
	NOME VARCHAR(100) PRIMARY KEY,
	TEMPOMEDIO VARCHAR(100) NOT NULL,
	PRECO FLOAT NOT NULL
);
CREATE TABLE ATENDIMENTO(
	ID SERIAL PRIMARY KEY,
	ATENDENTE VARCHAR(100),
	SERVICO VARCHAR(100),
	CLIENTE VARCHAR(50),
	HORAINICIO LOCALTIME,
	HORAFIM LOCALTIME,
	DATA LOCALDATE,
	CONFIRMADO BOOLEAN,
	FOREIGN KEY(ATENDENTE) REFERENCES ATENDENTE(NOME),
	FOREIGN KEY(SERVICO) REFERENCES SERVICO(NOME),
	FOREIGN KEY(CLIENTE) REFERENCES USUARIO(EMAIL)
);
CREATE TABLE AVALIACAO(
	ID SERIAL PRIMARY KEY,
	ATENDENTE VARCHAR(100),
	TEMPOATENDIMENTO INT NOT NULL,
	QUALIDADEATENDIMENTO INT NOT NULL,
	QUALIDADEAMBIENTE INT NOT NULL,
	QUALIDADESERVICO INT NOT NULL,
	FOREIGN KEY(ATENDENTE) REFERENCES ATENDENTE(NOME)
);
